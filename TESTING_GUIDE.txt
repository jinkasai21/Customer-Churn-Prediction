print(f"   ✓ Target: {y_proc.shape}")

print("\n3. Loading trained model...")

    model = pickle.load(f)
print(f"   ✓ Model type: {type(model).__name__}")

# 4. Test predictions
print("\n4. Testing predictions...")
y_pred = model.predict(X_proc)
y_proba = model.predict_proba(X_proc)
print(f"   ✓ Predictions shape: {y_pred.shape}")
print(f"   ✓ Probabilities shape: {y_proba.shape}")

print("\n5. Evaluating model...")
y_proc_flat = y_proc.values.ravel()
accuracy = accuracy_score(y_proc_flat, y_pred)
precision = precision_score(y_proc_flat, y_pred)
recall = recall_score(y_proc_flat, y_pred)
f1 = f1_score(y_proc_flat, y_pred)
roc_auc = roc_auc_score(y_proc_flat, y_proba[:, 1])
print(f"   ✓ Accuracy:  {accuracy:.4f}")
print(f"   ✓ Precision: {precision:.4f}")
print(f"   ✓ Recall:    {recall:.4f}")
print(f"   ✓ F1-Score:  {f1:.4f}")
print(f"   ✓ ROC-AUC:   {roc_auc:.4f}")
